\begin{tikzpicture}[scale=3]
  \coordinate (A) at (1cm,0);
  \coordinate (B) at (0,0);
  \coordinate (C) at (30:1cm);

  \filldraw [gray] 
    (A) circle [radius=1pt]
    (B) circle [radius=1pt]
    (C) circle [radius=1pt];

  % pic is not scaled.
  \draw (A) -- (B) -- (C)
        pic [draw=green!50!black, fill=green!20, angle radius=9mm,
             "$\alpha$"] {angle = A--B--C};

\end{tikzpicture}
